시간이 정말 걸리는 프로그램

#Add up the number of creatures in a data file.
filename = 'steve-2012.txt'
reader = open(filename, 'r')
total = 0
for line in reader:
    if not (line.startswith('#') or line.startswith('Date')):
        total = total + int(line.strip().split(',')[2])
reader.close()
print ('total is', total)
