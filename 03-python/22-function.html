def rescale(ivalue):
    ''' 입력값으로 벡터를 받고, 0.0에서 1.0사이의 범위로 조정되게 해당 벡터를 반환한다.
        사용례 rescale([1,2,3]) --> [0.0, 0.5, 1.0]'''
    ivalue.sort()
    H = ivalue[-1]
    L = ivalue[0]
    for item in range(len(ivalue)):
        ivalue[item] = float((ivalue[item]-L)/(H-L))
    return ivalue

help (rescale)

rslt = rescale([1,5,3])

print (rslt)
