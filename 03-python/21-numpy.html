import numpy as np
from matplotlib import pyplot as plt

data = np.loadtxt(fname='C:\_software-carpentry\python-novice-inflammation\data\inflammation-01.csv', delimiter=',')

plt.figure(figsize=(10.0, 3.0))

plt.subplot(3, 1, 1)
plt.ylabel('average')
plt.plot(data.mean(0))

plt.subplot(3, 1, 2)
plt.ylabel('max')
plt.plot(data.max(0))

plt.subplot(3, 1, 3)
plt.ylabel('min')
plt.plot(data.min(0))

plt.tight_layout()
plt.show()
