    def average(ivalue):
    assert len(ivalue) > 0
    assert type(ivalue) is list
    tot = 0
    ivalue.sort()
    for idx in range(len(ivalue)):
        tot += ivalue[idx]
        avg = float(tot)/len(ivalue)
    assert avg >= ivalue[0] and avg <= ivalue[1] 
    return avg
